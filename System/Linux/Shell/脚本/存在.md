```shell
myPath="/var/log/httpd/"
myFile="/var /log/httpd/access.log"
 
# 这里的-x 参数判断$myPath是否存在并且是否具有可执行权限
if [ ! -x "$myPath"]; then
mkdir "$myPath"
fi

# 判断路径
if [ ! -d "$myPath"]; then
mkdir "$myPath"
fi
 
# 判断文件
if [ ! -f "$myFile" ]; then
touch "$myFile"
fi

# 判断变量是否有值
if [ ! -n "$myVar" ]; then

exit 0
fi

if [ "$var1" = "$var2" ]; then
echo '$var1 eq $var2'

fi
```
